//query nr 1

select *
from prodotto p
where p.cod.prodotto=(select s.prodotto
					  from (select s.prodotto,count(*) as numero_vendite
							from scontrino s
							group by prodotto)
							having max(mumero_vendite));

//query nr4

select *
from fornitore for
where for.nome =( select f.fornitore
				 from (select f.fornitore, count(*) as numero_fatture
					   from fattura f
						goup by fornitore)
				group by fornitore
				having max(numero_fatture));

//query nr5

select p.categoria
from prodotto p join scontrino s
where sum(s.subTotale)= select max(sum(s1.subTotale))
						from scontrino s1
						group by s1.data
